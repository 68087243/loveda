【 商品销售统计 】
select productid,productname,namecn,sum(num) as num,price,sum(num*price) as amount,supplyname from my_order_detail where addtime between '2012-1-1' and '2015-1-1' group by productid order by productid
select a.productid,a.productname,a.namecn,sum(a.num) as num,a.price,sum(a.num*a.price) as amount,a.supplyname,b.sold,b.hits,b.status from my_order_detail as a inner join my_content as b on a.productid=b.id where a.addtime between '2012-1-1' and '2015-1-1' group by a.productid order by a.productid

【 销售额统计 】
select year(addtime) as 年,month(addtime) as 月,count(*) as 订单数,sum(amount) as 销售额 from `my_order` where addtime between '2012-1-1' and '2015-1-1' group by year(addtime),month(addtime) order by year(addtime),month(addtime)
select year(addtime) as 年,month(addtime) as 月,day(addtime) as 日,count(*) as 订单数,sum(amount) as 销售额 from `my_order` where addtime between '2012-1-1' and '2015-1-1' group by year(addtime),month(addtime),day(addtime) order by year(addtime),month(addtime),day(addtime)
